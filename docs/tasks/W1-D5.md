## 步骤

### 1. [CODE] 修改 `player.gd` — 自动瞄准

加一个找最近敌人的方法，替换鼠标瞄准：

```gdscript
func _physics_process(_delta: float) -> void:
	var direction := Input.get_vector("move_left", "move_right", "move_up", "move_down")
	velocity = direction * walk_speed
	move_and_slide()
	if Input.is_action_pressed("fire") and weapon != null:
		var target := _find_nearest_enemy()
		if target != null:
			weapon.try_fire(target.global_position)


func _find_nearest_enemy() -> Node2D:
	var enemies := get_tree().get_nodes_in_group("enemy")
	var best: Node2D = null
	var best_dist := INF
	for e in enemies:
		if not e is Node2D:
			continue
		var dist := global_position.distance_to(e.global_position)
		if dist < best_dist:
			best_dist = dist
			best = e
	return best
```

> 注意：D4 里 dummy 已经 `add_to_group("enemy")`，所以 `_find_nearest_enemy` 直接能找到。

### 2. [CODE] 创建 HUD 脚本 `src/scripts/ui/hud.gd`

```gdscript
class_name HUD
extends Control

@onready var hp_bar: TextureProgressBar = $TopBar/HpBar
@onready var hp_label: Label = $TopBar/HpBar/Label
@onready var ammo_label: Label = $TopBar/AmmoLabel
@onready var weapon_label: Label = $TopBar/WeaponLabel

var _player: Player
var _weapon: Weapon


func bind(player: Player) -> void:
	_player = player
	_weapon = player.weapon
	if _player.has_signal("hp_changed"):
		_player.hp_changed.connect(_on_hp_changed)
	_on_hp_changed(_player.hp, _player.max_hp)
	if _weapon and _weapon.config:
		weapon_label.text = _weapon.config.display_name


func _process(_delta: float) -> void:
	if _weapon == null:
		return
	var max_ammo: int = 0
	if _weapon.config:
		max_ammo = _weapon.config.mag_size
	ammo_label.text = "%d / %d" % [_weapon.current_ammo, max_ammo]


func _on_hp_changed(new_hp: int, max_hp: int) -> void:
	hp_bar.max_value = max_hp
	hp_bar.value = new_hp
	hp_label.text = "%d / %d" % [new_hp, max_hp]
```

## 常见坑

1. `get_tree().get_nodes_in_group("enemy")` 在 group 没注册过时返回空 — D4 必须在 dummy.gd 的 _ready 里 `add_to_group("enemy")`
2. TextureProgressBar 不写 texture_progress 会显示一片空白色 — 用 `tint_progress` 染色就够 MVP
3. TouchScreenButton 的 `pressed` 信号在 Godot 4 是 `pressed`，**不是** `button_pressed`
4. HUD bind 必须在 player._ready 之后 — `raid.gd._ready` 自然在子节点 _ready 之后调用，但如果你把 hud.bind 写在 hud.gd._ready 里直接找 player 会更安全
5. CanvasLayer 让 UI 不受 Camera2D 影响 — 永远把 UI 放 CanvasLayer 下
