### 3. [CODE] 创建虚拟摇杆 `src/scripts/ui/virtual_joystick.gd`

最小可用版本：手指按下 → 拖动 → 松开 reset。每帧把模拟摇杆方向写入 InputEventAction 注入到 input map。

```gdscript
class_name VirtualJoystick
extends Control

@export var radius: float = 100.0
@export var deadzone_normalized: float = 0.2

var _touch_index: int = -1
var _center: Vector2
var _knob_offset: Vector2 = Vector2.ZERO

@onready var base: TextureRect = $Base
@onready var knob: TextureRect = $Knob


func _ready() -> void:
	_center = base.global_position + base.size * 0.5


func _input(event: InputEvent) -> void:
	if event is InputEventScreenTouch:
		_handle_touch(event)
	elif event is InputEventScreenDrag and event.index == _touch_index:
		_handle_drag(event.position)
	elif event is InputEventMouseButton:
		# desktop fallback for testing
		if event.pressed and _is_in_base(event.position):
			_touch_index = 0
			_handle_drag(event.position)
		elif not event.pressed and _touch_index == 0:
			_release()
	elif event is InputEventMouseMotion and _touch_index == 0:
		_handle_drag(event.position)


func _handle_touch(event: InputEventScreenTouch) -> void:
	if event.pressed and _touch_index == -1 and _is_in_base(event.position):
		_touch_index = event.index
		_handle_drag(event.position)
	elif not event.pressed and event.index == _touch_index:
		_release()


func _handle_drag(pos: Vector2) -> void:
	var offset := pos - _center
	if offset.length() > radius:
		offset = offset.normalized() * radius
	_knob_offset = offset
	knob.position = base.position + base.size * 0.5 + offset - knob.size * 0.5
	_inject_actions()


func _release() -> void:
	_touch_index = -1
	_knob_offset = Vector2.ZERO
	knob.position = base.position + base.size * 0.5 - knob.size * 0.5
	_clear_actions()


func _is_in_base(pos: Vector2) -> bool:
	return (pos - _center).length() <= radius * 1.5  # generous touch area


func _inject_actions() -> void:
	var dir := _knob_offset / radius
	if dir.length() < deadzone_normalized:
		_clear_actions()
		return
	_set_action_strength("move_left", max(0.0, -dir.x))
	_set_action_strength("move_right", max(0.0, dir.x))
	_set_action_strength("move_up", max(0.0, -dir.y))
	_set_action_strength("move_down", max(0.0, dir.y))


func _clear_actions() -> void:
	for a in ["move_left", "move_right", "move_up", "move_down"]:
		Input.action_release(a)


func _set_action_strength(action: String, strength: float) -> void:
	if strength <= 0.0:
		Input.action_release(action)
		return
	if not Input.is_action_pressed(action):
		Input.action_press(action, strength)
	else:
		Input.action_press(action, strength)
```

## 常见坑

1. `move_and_slide()` 在 Godot 4 不接受参数 — 直接调用，velocity 自己设到 self.velocity
2. CharacterBody2D 必须有 CollisionShape2D 否则 move_and_slide 不工作
3. VirtualJoystick 用 `Input.action_press(action, strength)` 注入，**不要**用 `InputEventAction.parse_input_event`
4. CanvasLayer 包 UI 是为了让 UI 不受 Camera2D 影响（D3 加摄像机时不会跟着歪）
5. 触摸事件要用 `InputEventScreenTouch` / `InputEventScreenDrag`，鼠标用 `InputEventMouseButton` — 桌面 fallback 别忘了
